What is HEX and RGB?


HEX

A HEX Triplet (a.k.a: HEX color) is a six-digit three-byte hexadecimal number standardly used in HTML and CSS to represent colors. The first two digits represent the red value, the second two digits represent the green value, and the last two digits represent the blue value. Each digit can be a number from 0 to 9 or a letter from A to F. For example, #FF0000 is red, #00FF00 is green, and #0000FF is blue.


RGB

RGB stands for Red, Green, and Blue. It is a color model that uses red, green, and blue light to create all the colors on your computer screen. The RGB color model is an additive color model, which means that it describes how light is added together to create colors.

Here are some examples in different programming languages for HEX to RGB

Python: 

    def hex_to_rgb(value):
        value = value.lstrip('#')
        lv = len(value)
        return tuple(int(value[i:i + lv // 3], 16) for i in range(0, lv, lv // 3))
JavaScript: 

    function hexToRgb(hex) {
        var bigint = parseInt(hex, 16);
        var r = (bigint >> 16) & 255;
        var g = (bigint >> 8) & 255;
        var b = bigint & 255;
        return r + "," + g + "," + b;
    }
C++: 

    #include <string>
    #include <sstream>
    
    std::string hexToRgb(std::string hex)
    {
        unsigned int hexInt;
        std::stringstream ss;
        ss << std::hex << hex;
        ss >> hexInt;
        int r = (hexInt >> 16) & 0xFF;
        int g = (hexInt >> 8) & 0xFF;
        int b = (hexInt) & 0xFF;
        std::stringstream ss2;
        ss2 << r << "," << g << "," << b;
        return ss2.str();
    }
C#: 

    public static int[] HexToRgb(string hex)
    {
        hex = hex.Replace("#", "");
        int r = int.Parse(hex.Substring(0, 2), NumberStyles.HexNumber);
        int g = int.Parse(hex.Substring(2, 2), NumberStyles.HexNumber);
        int b = int.Parse(hex.Substring(4, 2), NumberStyles.HexNumber);
        return new int[] { r, g, b };
    }
Java: 

    public static int[] hexToRgb(String colorStr) {
        int r = Integer.valueOf(colorStr.substring(1, 3), 16);
        int g = Integer.valueOf(colorStr.substring(3, 5), 16);
        int b = Integer.valueOf(colorStr.substring(5, 7), 16);
        return new int[] { r, g, b };
    }
Ruby: 

    def hex_to_rgb(hex)
        hex = hex[1..-1]
        r = hex[0..1].hex
        g = hex[2..3].hex
        b = hex[4..5].hex
        [r, g, b]
    end
Go: 

    func hexToRgb(hex string) (int, int, int) {
        r, _ := strconv.ParseInt(hex[1:3], 16, 0)
        g, _ := strconv.ParseInt(hex[3:5], 16, 0)
        b, _ := strconv.ParseInt(hex[5:7], 16, 0)
        return int(r), int(g), int(b)
    }
Rust: 

    fn hex_to_rgb(hex: &str) -> (u8, u8, u8) {
        let r = u8::from_str_radix(&hex[1..3], 16).unwrap();
        let g = u8::from_str_radix(&hex[3..5], 16).unwrap();
        let b = u8::from_str_radix(&hex[5..7], 16).unwrap();
        (r, g, b)
    }
Swift: 

    func hexToRgb(hex: String) -> (Int, Int, Int) {
        let r = Int(hex[1...2], radix: 16)!
        let g = Int(hex[3...4], radix: 16)!
        let b = Int(hex[5...6], radix: 16)!
        return (r, g, b)
    }
Kotlin: 

    fun hexToRgb(hex: String): Triple {
        val r = hex.substring(1, 3).toInt(16)
        val g = hex.substring(3, 5).toInt(16)
        val b = hex.substring(5, 7).toInt(16)
        return Triple(r, g, b)
    }
